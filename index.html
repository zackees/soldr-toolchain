Resolve in Python

pip install git+https://github.com/zackees/manifest.json

python -c "
import json, urllib.request
from manifest_json import resolve_in_catalog
url = 'https://zackees.github.io/soldr-toolchain/zccache/manifest.json'
catalog = json.loads(urllib.request.urlopen(url).read())
asset = resolve_in_catalog(catalog, 'zccache',
    platform={'os': 'linux', 'arch': 'x86_64', 'libc': 'musl'},
    channel='latest-stable')
print(asset['urls'][0])
"

manifest-resolve <(curl -fsSL https://zackees.github.io/soldr-toolchain/zccache/manifest.json) \

    --platform os=linux,arch=x86_64,libc=musl \
